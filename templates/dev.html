const tx    = await sendMultipassPayment(1, destG1pub, null, email);
const ok    = await verifyMultipassPayment(tx.id);
alert(ok ? '✅ 1 ẐEN envoyé' : '⚠ Réessayez');

const phiA = Phi2X.computePhaseFromForm('1988-03-15', '12:00', 2.35);
